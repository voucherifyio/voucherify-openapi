<h2>Location object</h2>
<table>
  <thead>
    <tr>
      <th>Attributes</th>
      <th>Description</th>
    </tr>
  </thead>
  <tbody>
    <tr>
      <td>
        id
        <br />
        <code>string</code>
      </td>
      <td>
        <p>Unique location ID, assigned by the Voucherify API.</p>
        <strong>Example:</strong>
        <p>loc_NoMGXmHO9OUs7iz9mGWpamma</p>
      </td>
    </tr>
    <tr>
      <td>
        object
        <br />
        <code>string</code>
      </td>
      <td>
        <p>
          The type of the object represented by JSON. This object stores information about a
          <code>location</code>.
        </p>
        Available values:
        <code>location</code>
      </td>
    </tr>
    <tr>
      <td>
        name
        <br />
        <code>string</code>
      </td>
      <td>
        <p>Location name.</p>
      </td>
    </tr>
    <tr>
      <td>
        shape
        <br />
        <code>object</code>
      </td>
      <td>
        <p>Defines the shape and boundaries of the location.</p>
        One of:
        <a href="#circle">Circle</a>,
        <a href="#polygon">Polygon</a>
      </td>
    </tr>
    <tr>
      <td>
        created_at
        <br />
        <code>string</code>
      </td>
      <td>
        <p>Timestamp representing the date and time when the location was created. The value is shown in the ISO 8601 format.</p>
        <strong>Example:</strong>
        <p>2022-02-14T15:12:06.817Z</p>
      </td>
    </tr>
    <tr>
      <td>
        updated_at
        <br />
        <code>string</code>,
        <code>null</code>
      </td>
      <td>
        <p>Timestamp representing the date and time when the location was updated. The value is shown in the ISO 8601 format.</p>
        <strong>Example:</strong>
        <p>2022-03-14T15:12:06.817Z</p>
      </td>
    </tr>
  </tbody>
</table>

    <tr>

    </tr>
    <tr>
      <td>
        format
        <br />
        <code>string</code>
      </td>

        </p>
        Available values:
        <code>distance</code>
      </td>
    </tr>
    <tr>
      <td>
        distance
        <br />
        <code>object</code>
      </td>
      <td>

        <table>
          <thead>
            <tr>
              <th>Attributes</th>
              <th>Description</th>
            </tr>
          </thead>
          <tbody>
            <tr>
              <td>
                center
                <br />
                <code>string</code>
              </td>
              <td>

                <p>geo:40.79372699823857,-74.15092132694554</p>
              </td>
            </tr>
            <tr>
              <td>
                radius
                <br />
                <code>string</code>
              </td>
              <td>
                <p>Defines the radius of the circle.</p>
              </td>
            </tr>
          </tbody>
        </table>
      </td>
    </tr>
  </tbody>
</table>
