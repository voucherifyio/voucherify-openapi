        <br>

        <p>2022-08-11T14:49:22.586Z</p>
      </td>
    </tr>
    <tr>
      <td>
        updated_at
        <br>
        <code>string</code>,
        <code>null</code>
      </td>

        <strong>Example:</strong>
        <p>2022-08-11T16:01:34.885Z</p>
      </td>
    </tr>
    <tr>
      <td>
        object
        <br>
        <code>string</code>
      </td>
      <td>
        <p>The type of the object represented by the JSON. This object stores information about the reward assignment.</p>
        Available values:
        <code>reward_assignment</code>
      </td>
    </tr>
    <tr>
      <td>
        related_object_id
        <br>
        <code>string</code>
      </td>
      <td>
        <p>Related object ID to which the reward was assigned.</p>
        <strong>Example:</strong>
        <p>camp_wciTvaOfYmAa3EmIIW3QpXXZ</p>
      </td>
    </tr>
    <tr>
      <td>
        related_object_type
        <br>
        <code>string</code>
      </td>
      <td>
        <p>Related object type to which the reward was assigned.</p>
        Available values:
        <code>campaign</code>

    </tr>
